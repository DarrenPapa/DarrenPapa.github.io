How to Program in DPL

Hello World!

&use {std/text_io.py}
io:println "Hello, world!"

Using Variables

&use {std/text_io.py}
set this 90
io:println :this

Formatted Strings

&use {std/text_io.py}
set hello_world "Hello, world!"
io:println '${hello_world}'
io:println [fast-format "${hello_world}"]

Functions

# fn fn_name [!arg1 arg2 ... argN]
fn greet [!name]
    io:println 'Hello ${name}!'
end

greet "Darren"

Conditionals

if cond
    ... true block
end

match value
    as alias
    with literal
        ... body
        ... fallthrough
    end
    case cond
        ... body
        ... fallthrough
    end
    default
        ... body
    end
end

Loops

loop
    ... loops indefinitely
end

loop num
    ... loops 'num' number of types
end

for name in iterable
    ... loops through the given iterable
end

while cond
    ... loop while cond stays true
end

Classes

&use {std/text_io.py}

object Person
method  Person.new [!name]
    new :self tmp
    set tmp.name :name
    return :tmp
end
method Person.greet [!name]
    io:println 'Hello ${name} Im ${self.name}'
end

catch [!person] Person.new "Andrew"
person.greet "Carl"
